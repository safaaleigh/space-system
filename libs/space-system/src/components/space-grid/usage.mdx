# space-grid

import {
  applyPolyfills,
  defineCustomElements,
} from '../../../../../dist/libs/space-system/loader';

export const Container = (props) => (
  <space-grid
    {...props}
    style={{
      borderRadius: '.5rem',
      padding: '1rem',
      backgroundColor: 'grey',
      ...props.style,
    }}
  ></space-grid>
);

## Basic

export const Basic = () => {
  applyPolyfills().then(() => {
    defineCustomElements();
  });
  return (
    <space-grid>
      <Container />
      <Container />
    </space-grid>
  );
};

```jsx
<space-grid>
  <space-grid />
  <space-grid />
</space-grid>
```

<Basic />
